  final String activity;

  3. Create an API Service:
    In lib/services/api_service.dart, create a class to handle API requests:

    dart
    Copy
    import 'dart:convert';
import 'package:http/http.dart' as http;
import '../models/fitness_data.dart';

class ApiService {
  final String apiKey = 'YOUR_NUTRITIONIX_API_KEY';
  final String appId = 'YOUR_NUTRITIONIX_APP_ID';
  final String baseUrl = 'https://trackapi.nutritionix.com/v2/natural/exercise';

  Future<List<FitnessData>> fetchFitnessData(String query) async {
    final response = await http.post(
      Uri.parse(baseUrl),
      headers: {
        'Content-Type': 'application/json',
        'x-app-id': appId,
        'x-app-key': apiKey,
      },
      body: jsonEncode({'query': query}),
    );

    if (response.statusCode == 200) {
      final data = jsonDecode(response.body);
      return (data['exercises'] as List)
          .map((e) => FitnessData.fromJson(e))
          .toList();
    } else {
      throw Exception('Failed to load data');
    }
  }
}

    import 'package:flutter/material.dart';
import '../models/fitness_data.dart';

class FitnessProvider with ChangeNotifier {
  List<FitnessData> _fitnessData = [];
  bool _isLoading = false;

  List<FitnessData> get fitnessData => _fitnessData;
  bool get isLoading => _isLoading;

  final ApiService _apiService = ApiService();

  Future<void> fetchData(String query) async {
    _isLoading = true;
    notifyListeners();

    try {
      _fitnessData = await _apiService.fetchFitnessData(query);
    } catch (e) {
      print(e);
    } finally {
      _isLoading = false;
      notifyListeners();
    }
  }
}

Step 3: Set Up OAuth 2.0 Authentication

Fitbit uses OAuth 2.0 for authentication. We'll use the flutter_web_auth package to handle the OAuth flow.

  1. Create a Helper Class for Authentication:
    In lib/services/auth_service.dart, create a class to handle OAuth:

    dart
    Copy
    import 'package:flutter_web_auth/flutter_web_auth.dart';
import 'dart:convert';
import 'package:http/http.dart' as http;

class AuthService {
  static const String clientId = 'YOUR_CLIENT_ID'; // Replace with your Fitbit Client ID
  static const String clientSecret = 'YOUR_CLIENT_SECRET'; // Replace with your Fitbit Client Secret
  static const String redirectUri = 'http://localhost:8080';
  static const String authorizationEndpoint =
      'https://www.fitbit.com/oauth2/authorize';
  static const String tokenEndpoint = 'https://api.fitbit.com/oauth2/token';

  Future<String> authenticate() async {
    final url =
        '$authorizationEndpoint?response_type=code&client_id=$clientId&redirect_uri=$redirectUri&scope=activity%20heartrate%20sleep';

    // Start the OAuth flow
    final result = await FlutterWebAuth.authenticate(
      url: url,
      callbackUrlScheme: 'http',
    );

    // Extract the authorization code from the result
    final code = Uri.parse(result).queryParameters['code'];
    if (code == null) {
      throw Exception('Authorization code not found');
    }

    return code;
  }

  Future<Map<String, dynamic>> getAccessToken(String code) async {
    final response = await http.post(
      Uri.parse(tokenEndpoint),
      headers: {
        'Content-Type': 'application/x-www-form-urlencoded',
        'Authorization': 'Basic ${base64Encode(utf8.encode('$clientId:$clientSecret'))}',
      },
      body: {
        'client_id': clientId,
        'grant_type': 'authorization_code',
        'redirect_uri': redirectUri,
        'code': code,
      },
    );

    if (response.statusCode == 200) {
      return jsonDecode(response.body);
    } else {
      throw Exception('Failed to get access token');
    }
  }
}

In Flutter, you can increase the height of the AppBar by using the toolbarHeight property. This property allows you to specify a custom height for the AppBar. Here's how you can do it:

Step-by-Step Guide

  1. Default AppBar:
    By default, the AppBar has a height of 56 logical pixels (on mobile devices). Here's an example of a default AppBar:

    dart
    Copy
    Scaffold(
  appBar: AppBar(
    title: Text('Default AppBar'),
  ),
  body: Center(
    child: Text('Hello, World!'),
  ),
);

  2. Increase AppBar Height:
    To increase the height, use the toolbarHeight property. For example, to set the height to 100 pixels:

    dart
    Copy
    Scaffold(
  appBar: AppBar(
    title: Text('Custom Height AppBar'),
    toolbarHeight: 100, // Set the height here
  ),
  body: Center(
    child: Text('Hello, World!'),
  ),
);

  3. Add Custom Content:
    If you want to add more content (e.g., buttons, images, or text) to the AppBar, you can use the flexibleSpace property. For example:

    dart
    Copy
    Scaffold(
  appBar: AppBar(
    title: Text('Custom Height AppBar'),
    toolbarHeight: 100, // Set the height here
    flexibleSpace: Center(
      child: Column(
        mainAxisAlignment: MainAxisAlignment.center,
        children: [
          Text('This is a custom AppBar'),
          ElevatedButton(
            onPressed: () {},
            child: Text('Click Me'),
          ),
        ],
      ),
    ),
  ),
  body: Center(
    child: Text('Hello, World!'),
  ),
);

  4. Use a Custom Widget:

    dart

        child: Column(
          mainAxisAlignment: MainAxisAlignment.center,
          children: [
            Text('Custom AppBar'),
            ElevatedButton(
              onPressed: () {},
              child: Text('Click Me'),
            ),
          ],
        ),
      ),
    ),
  ),
  body: Center(
    child: Text('Hello, World!'),
  ),
);
